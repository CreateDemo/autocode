SELECT  cb.F_CONTPRODID ITEM_ID,
        cb.CONID CONID,
        CPB.NAME NAME,
        F.ORGID FILIALE,
        F.ORGNAME ORGNAME,
        CB.ID BILL_ID,
        CB.F_BILLNO BILLNO,
        CCIDSUB.TERMTIME TERM_DAY,
        CB.F_STATE FSTATUS,
        CB.CONNO CONNO,
        CPB.NAME PRODUCT_TYPE,
        cb.ITEM_PAYTIME PAYTIME,
        cb.SERVICE_NO SERVICE_NO,
        CASE WHEN CCOP.BUSI_APPLICATION_NUMBER='' THEN CB.batchnumber WHEN CCOP.BUSI_APPLICATION_NUMBER IS NULL THEN CB.batchnumber ELSE CCOP.BUSI_APPLICATION_NUMBER END SN,
        CCOP.BRANDDESC BRANDDESC
FROM  W_CRM_BILL_INFO CB
LEFT JOIN w_crm_bill_info_sub t on t.BILL_ID = CB.ID
LEFT JOIN CRM_CON_ITEM_DETAIL_SUB CCIDSUB ON cb.F_CONTPRODID=CCIDSUB.CONDETAILID
LEFT JOIN CRM_PRODUCT_BASICINFO CPB ON CPB.ID = cb.GOODSID
LEFT JOIN CRM_PRODUCT_CATEGORY CPC ON CPB.CATEGORYID = CPC.ID
LEFT JOIN CRM_CON_OPERDATA CCOP on CCOP.serviceno = CB.SERVICE_NO
LEFT JOIN CRM_CON_SUBJECT CCS ON CCS.SERVICENO=cb.SERVICE_NO
LEFT JOIN CRM_CUST CUST ON cb.CUSTID = CUST.ID
LEFT JOIN sys_user_pos PO ON PO.userid =cb.follow_user_id
LEFT JOIN SYS_ORG E ON E.ORGID =PO.orgid
LEFT JOIN SYS_ORG F ON E.ORGSUPID =F.ORGID
LEFT JOIN W_CRM_TRADEMARK_PRO P ON P.id = t.LAST_PROBLEM_ID
where  CPC.PRODUCT_TYPE = 'sb'  AND PO.isprimary=1 AND PO.ISDELETE=0  AND CB.F_STATE !=10 AND T.bill_id IS NOT NULL LIMIT 1
